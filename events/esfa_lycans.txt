namespace = lycans

lycans.10 = {
	type = country_event
	hidden = yes
	
	trigger = {
		current_ruler = {
			has_trait = werewolf
		}
		religion = night_pantheon
		primary_culture = frostmoon
	}
	
	immediate = { 
		random_character = {
			limit = {
				employer = { 
					has_land = yes
				}
				is_alive = yes
				is_adult = yes
				NOT = {	has_trait = werewolf }
			}
			save_scope_as = candidate_gov
		}
		trigger_event = {
			id = lycans.1
		}
	}
	
}

lycans.4 = {
	type = country_event
	hidden = yes
	
	trigger = {
		religion = night_pantheon
		primary_culture = frostmoon
	}
	
	immediate = {
		every_character = {
			limit = {
				is_alive = yes
				OR = {
					exists = father
					exists = mother
				}
			}
			if = {
				limit = {
					OR = {
						father = {
							has_trait = werewolf
							is_ruler = yes
						}
						mother = {
							has_trait = werewolf
							is_ruler = yes
						}
					}
					NOT = {	has_trait = full_vampire }
				}
				save_scope_as = childofwolf
				trigger_event = {
					id = lycans.5
				}
			}
		}
	}
}
