	type = country_event
	hidden = yes
	
	trigger = {
		tag = VBM
	}
	
	immediate = { 
		random_character = {
			limit = {
				employer = { 
					has_land = yes
				}
				is_alive = yes
				is_adult = yes
				NOT = {	has_trait = lower_vampire }
				NOT = {	has_trait = half_vampire }

				NOT = {	has_trait = werewolf }
			}
			save_scope_as = candidate_gov
		}
		trigger_event = {
			id = vampirism.1
		}
	}
	
}
