namespace = bloodlines_setup_completion

bloodlines_setup_completion.1 = {
	type = country_event
	hidden = yes
	
	immediate = {
		every_character = {
			limit = {
				is_alive = yes
				OR = {
					exists = father
					exists = mother
				}
			}
			if = {
				limit = {
					exists = father
					father = {
						has_trait = davidids
					}
					NOT = {
						has_trait = davidids
					}
				}
				add_trait = davidids
			}
			
			if = {
				limit = {
					exists = mother
					mother = {
						has_trait = davidids
					}
					NOT = {
						has_trait = davidids
					}
				}
				add_trait = davidids	
			}
			
		}
	}
}
