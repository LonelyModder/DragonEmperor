namespace = esfa_traits

#Trait inheritance events
esfa_traits.1 = {		
	type = character_event
	hidden = yes

	immediate = {
		#Racial traits
		if = {
			limit = {
				mother = { has_trait = altmer }
			}
			add_trait = altmer
		}	
		else_if = { 
			limit = {
				mother = { has_trait = bosmer }
			}
			add_trait = bosmer
		}	
		else_if = { 
			limit = {
				mother = { has_trait = nede }
			}
			add_trait = nede
		}
		else_if = { 
			limit = {
				mother = { has_trait = goblin }
			}
			add_trait = goblin
		}
		else_if = { 
			limit = {
				mother = { has_trait = ayleid }
			}
			add_trait = ayleid
		}
		#Altmer racism
		if = {
			limit = {
				has_trait = altmer
				OR = {
					OR = {
						AND = {
							father = { has_trait = pure }
							mother = { has_trait = impure }
						}
						AND = {
							father = { has_trait = impure }
							mother = { has_trait = pure }
						}
					}
					AND = {
						father = { 
							OR = { 
								has_trait = ayleid
								has_trait = bosmer 	
								has_trait = nede 
								has_trait = goblin 
							} 
						}
						mother = { has_trait = altmer }
					}
				}	
			}
			add_trait = impure
		}
		else_if = { 
			limit = {
				has_trait = altmer
			}
			add_trait = pure
		}
	}
					
}
