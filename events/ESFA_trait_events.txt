namespace = esfa_traits

#Trait inheritance events
esfa_traits.1 = {		
	type = character_event
	hidden = yes

	immediate = {
		#Racial traits
		if = {
			limit = {
				OR = {
					father = { has_trait = minotaur }
					mother = { has_trait = minotaur }
				}
			}
			add_trait = minotaur
		}
		if = {
			limit = {
				mother = { has_trait = altmer }
			}
			add_trait = altmer
		}	
		
		else_if = { 
			limit = {
				mother = { has_trait = bosmer }
			}
			add_trait = bosmer
		}	
		else_if = { 
			limit = {
				mother = { has_trait = nede }
			}
			add_trait = nede
		}
		else_if = { 
			limit = {
				mother = { has_trait = goblin }
			}
			add_trait = goblin
		}
		else_if = { 
			limit = {
				mother = { has_trait = ayleid }
			}
			add_trait = ayleid
		}
		else_if = { 
			limit = {
				mother = { has_trait = orc }
			}
			add_trait = orc
		}
		else_if = { 
			limit = {
				mother = { has_trait = imga }
			}
			add_trait = imga
		}
		else_if = { 
			limit = {
				mother = { has_trait = khajiit }
			}
			add_trait = khajiit
		}
		else_if = { 
			limit = {
				mother = { has_trait = argonian }
			}
			add_trait = argonian
		}
		#Altmer racism
		if = {
			limit = {
				has_trait = altmer
				OR = {
					OR = {
						AND = {
							father = { has_trait = pure }
							mother = { has_trait = impure }
						}
						AND = {
							father = { has_trait = impure }
							mother = { has_trait = pure }
						}
					}
					AND = {
						father = { 
							OR = { 
								has_trait = ayleid
								has_trait = bosmer 	
								has_trait = nede 
								has_trait = goblin 
								has_trait = orc
								has_trait = khajiit
								has_trait = argonian
								has_trait = imga
							} 
						}
						mother = { has_trait = altmer }
					}
				}	
			}
			add_trait = impure
		}
		else_if = { 
			limit = {
				has_trait = altmer
			}
			add_trait = pure
		}
	}
					
}
