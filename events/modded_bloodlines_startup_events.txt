namespace = bloodlines_startup

bloodlines_startup.1 = { #Aranas

	type = country_event
	hidden = yes
	
	trigger = {
		tag = AAA
	}
	
	immediate = {
		every_character = {
			limit = {
				has_same_family = char:1
			}
			add_trait = aranas
		
		}
		hidden_effect = {
			trigger_event = { id = bloodlines_setup_completion.1 }
		}
	
	}

}
