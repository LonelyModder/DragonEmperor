unificationwars_mission_1.5 = {

	picture = nordic_feud

	trigger = {
		tag=SAG
	}
	
	option = { 
		name = "unificationwars_mission_1.5.a"
		c:SAI = {
			every_owned_province = { 
				limit = { 
					has_owner = yes 
				}
				set_conquered_by = c:SAG
			}
		}
		create_character = {
			first_name = "Harald"
			family_name = "Battleborn"
			no_stats = yes
			add_trait="nord"
			add_trait="companion"
			culture = westholder
			religion = nordic_pantheon
			age = 13
			good_martial_effect = yes
			good_charisma_effect = yes
			average_finesse_effect = yes
			average_zeal_effect = yes
			add_gold=50
		}
		create_character = {
			first_name = "Otto"
			family_name = "Greymane"
			no_stats = yes
			add_trait="nord"
			add_trait="companion"

			good_charisma_effect = yes

			add_gold=50
		}
	}
}
