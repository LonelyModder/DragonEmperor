ESFA_on_create_character_effect = {
	if = {
		limit = {
			has_culture_group = altmer
			NOT = { has_trait = altmer }
			NOT = { has_trait = pure }
		}
		add_trait = altmer
		add_trait = pure
	}
	else_if = {
		limit = {
			has_culture_group = bosmer
		}
		add_trait = bosmer
	}
	else_if = {
		limit = {
			has_culture_group = ayleid
			NOT = { has_trait = ayleid }
		}
		add_trait = ayleid
	}
	else_if = {
		limit = {
			has_culture_group = goblin
			NOT = { has_trait = goblin }
		}
		add_trait = goblin
	}
	else_if = {
		limit = {
			has_culture_group = nede
			NOT = { has_trait = nede }
		}
		add_trait = nede
	}
	else_if = {
		limit = {
			has_culture_group = orcish
			NOT = { has_trait = orc }
		}
		add_trait = orc
	}
	else_if = {
		limit = {
			has_culture_group = minotaurish
			NOT = { has_trait = minotaur }
		}
		add_trait = minotaur
	}
}
