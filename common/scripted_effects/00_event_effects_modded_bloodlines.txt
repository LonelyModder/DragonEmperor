
pass_down_traits_effect = {
	scope:newborn = {
		if = {
			limit = {
				exists = father
				OR = {
					father = {
						has_trait = davidids
					}
				}
			}
			if = {
				limit = {
					father = {
						has_trait = davidids
					}
					NOT = {
						has_trait = davidids
					}
				}
				add_trait = davidids
			}
			
		}
		if = {
			limit = {
				exists = mother
				mother = {
					has_trait = davidids
				}
				NOT = {
					has_trait = davidids
				}
			}
			add_trait = davidids

		}
				
	}
}

bastard_major_bloodline_effect = {
	if = {
		limit = {
			scope:bastard_ruler = {
				has_trait = davidids
			}
		}
		add_trait = davidids
	}
	
}
